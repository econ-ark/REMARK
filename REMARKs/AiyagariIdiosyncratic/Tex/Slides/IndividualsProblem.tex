\begin{frame}{Solve the Model}
	The decision rule can be written as: 
	\begin{align}
	\hat{a}_{t+1}=A(z_t,\phi,w,r)
	\end{align}
	
	And the law of transition would be:
	\begin{align}
	z_{t+1}=wl_{t+1}+(1+r)A(z_t,\phi,w,r)-r\phi
	\end{align}
	
\end{frame}
